\begin{algorithm}[H]
    \caption{Solution reconstruction}
    \label{reconstruction}
    \begin{algorithmic}[1]
        \Require{A set $V^L$ of tuples $(S, K)$, where $S_l$ is a set of solution-submatrices, and $K_l$ is a set of the costs}
        \Procedure{Reconstruct}{$V^L, l$} \Comment{Where $l$ is the number of submatrices}
            \State Set $P := \{1,...,l\}$
            \State Set $Q := \{Q_1,...Q_l\}$
            \For{$i \gets P$}
                \State Set $j := i$
                \While{$j < l$}
                    \State Set $j := j + 1$
                \EndWhile
            \EndFor
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
